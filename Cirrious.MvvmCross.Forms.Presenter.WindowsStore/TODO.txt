In order to integrate FormsPresenters plug-in correctly, please follow these steps:

1. Modify Windows Store project's MainPage.xaml.cs ctor. in the following way:

    public MainPage()
    {
        this.InitializeComponent();

        var presenter = Mvx.Resolve<IMvxViewPresenter>() as MvxFormsWindowsStorePagePresenter;
        LoadApplication(presenter.MvxFormsApp);
    }

2. Add the following piece of code to Setup.cs from Windows Phone project:

    protected override IMvxWindowsViewPresenter CreateViewPresenter(IMvxWindowsFrame rootFrame)
    {
        Forms.Init(null);

        var xamarinFormsApp = new MvxFormsApp();
        var presenter = new MvxFormsWindowsStorePagePresenter(rootFrame, xamarinFormsApp);
        Mvx.RegisterSingleton<IMvxViewPresenter>(presenter);

        return presenter;
    }
